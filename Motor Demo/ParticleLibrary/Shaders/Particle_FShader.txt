#version 330 core
in vec2 texCoords;
in vec4 particleColor;

out vec4 FragColor;

uniform sampler2D sprite;
uniform int useTexture;

void main()
{
	vec4 tex = texture(sprite, texCoords);
	if(tex.a < 0.1 && useTexture == 1)
		discard;

	if(useTexture == 1)
		FragColor = tex * particleColor;
	else
		FragColor = particleColor;
}
