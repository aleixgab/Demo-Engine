#version 330 core
in vec2 texCoords;
in vec4 particleColor;
in float lifeNorm;

out vec4 FragColor;

uniform sampler2D sprite;
uniform float colorPercent;

void main()
{
	vec4 tex = texture(sprite, texCoords);
	if(colorPercent < 1.0 && tex.a < 0.1)
		discard;

    FragColor = mix(tex, particleColor, colorPercent);
}
